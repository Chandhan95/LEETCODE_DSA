class Node:
    def __init__(self,data):
        self.data=data
        self.next=None

class Stack:
    def __init__(self):
        self.top=None
    def push(self,x):
        new_node=Node(x)
        new_node.next=self.top
        self.top=new_node
    def pop(self):
        if self.top is None:
            print("Empty")
            return None
        val=self.top.data
        self.top=self.top.next
        return val
    def peek(self):
        if self.top is None:
            print("Empty")
        val=self.top.data
        return val
    def isempty(self):
        return self.top is None
    def display(self):
        current=self.top
        if current is None:
            print("Empty")
        print("Stack elements are:")
        while current:
            print(current.data,end=" ")
            current=current.next
        print()
s=Stack()
s.push(5)
s.push(10)
s.push(18)
s.display()
s.pop()
s.display()
print(s.peek())
print(s.isempty())
